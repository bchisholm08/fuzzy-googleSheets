I used this code to help with creating a bug log in google sheets. It's a combination of JS and HTML 

First file is recordBug.gs

This contains two functions, recordBug and submitBugData. The first function is mostly a helper function. 
The second script does the heavy lifting. 
```javascript

// open the HTML form in dialog 
function recordBug() {
  const html = HtmlService.createHtmlOutputFromFile('BugForm')
    .setWidth(500)  
    .setHeight(400); 
  SpreadsheetApp.getUi().showModalDialog(html, 'Record a Bug');
}

// process interface data 
function submitBugData(description, notes, resolved) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  
  // insert  new row 
  sheet.insertRowBefore(2);
   
  const currentDateTime = new Date();
  sheet.getRange('A2').setValue(currentDateTime);
  
  // place inputs 
  sheet.getRange('B2').setValue(description);
  sheet.getRange('C2').setValue(notes);
    
  const checkboxCell = sheet.getRange('D2');
  const checkboxRule = SpreadsheetApp.newDataValidation()
    .requireCheckbox()
    .build();
  checkboxCell.setDataValidation(checkboxRule);

  // checkbox value 
  checkboxCell.setValue(resolved);

  // checkbox format
  const ruleUnchecked = SpreadsheetApp.newConditionalFormatRule()
    .whenFormulaSatisfied('=NOT($D2)')
    .setBackground('#FFCCCC')  
    .setRanges([checkboxCell])
    .build();
    
  const ruleChecked = SpreadsheetApp.newConditionalFormatRule()
    .whenFormulaSatisfied('=$D2')
    .setBackground('#CCFFCC')  
    .setRanges([checkboxCell])
    .build();
  
  const rules = sheet.getConditionalFormatRules();
  rules.push(ruleUnchecked, ruleChecked);
  sheet.setConditionalFormatRules(rules);
}


```
This last script is the HTML for the dialogue box, mostly generated by ChatGPT since I am no good with html

```html
<!DOCTYPE html>
<html>
  <head>
    <base target="_top">
    <style>
      body {
        background-color: #f0f0f0; /* Light gray background */
        font-family: Arial, sans-serif;
      }
      input[type="text"], textarea {
        width: 90%;
        padding: 8px;
        font-size: 16px;
      }
      textarea {
        height: 100px;
      }
      button {
        padding: 10px 20px;
        font-size: 16px;
      }
    </style>
  </head>
  <body>
    <form>
      <label for="bugDescription">Description of Bug:</label><br>
      <textarea id="bugDescription" name="bugDescription"></textarea><br><br>
      
      <label for="otherNotes">Other Notes:</label><br>
      <textarea id="otherNotes" name="otherNotes"></textarea><br><br>
      
      <input type="checkbox" id="resolved" name="resolved">
      <label for="resolved">Resolved?</label><br><br>
      
      <button type="button" onclick="submitBug()">Done</button>
    </form>
    <script>
      function submitBug() {
        const description = document.getElementById('bugDescription').value;
        const notes = document.getElementById('otherNotes').value;
        const resolved = document.getElementById('resolved').checked;
        
        google.script.run.withSuccessHandler(google.script.host.close)
          .submitBugData(description, notes, resolved);
      }
    </script>
  </body>
</html>

```
